View: vw_GameReport
SELECT GameId,
       AwayTeam.Name AS AwayTeam,
       HomeTeam.Name AS HomeTeam,
       f.Name AS Field,
       ChosenScheduleTime,
       League.Name AS League
  FROM Game
       INNER JOIN
       Team AwayTeam ON AwayTeam.TeamId = AwayTeamId
       INNER JOIN
       Team HomeTeam ON HomeTeam.TeamId = HomeTeamId
       INNER JOIN
       Field f ON f.FieldId = Game.FieldId
       INNER JOIN
       League ON League.LeagueId = Game.LeagueId
 ORDER BY League.LeagueId,
          ChosenScheduleTime ASC


View: vw_TeamGameCounts
SELECT TeamId, LeagueId, COUNT(*) AS GameCount
FROM (
    SELECT HomeTeamId as TeamId, LeagueId FROM game
    UNION ALL
    SELECT AwayTeamId as TeamId, LeagueId FROM game
) AS all_games
GROUP BY LeagueId, TeamId
order by LeagueId
